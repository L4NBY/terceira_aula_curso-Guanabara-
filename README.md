# terceira_aula_curso-Guanabara-
Terceira aula curso Guanabara em JS &lt; JavaScript 
```

<!DOCTYPE html>
<html>

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>terceira aula do curso do Guanabara</title>
</head>

<body>
<script>
  var n1 = Number.parseInt(window.prompt("digite um número: "))
  var n2 = Number.parseInt(window.prompt("digite outro número: ")) // estou jogando o valor do número digitado para número inteiro
  
  var s = n1+n2 //estou fumando as duas variáveis de número inteiro .
  window.alert("o valor entre as somas foi , "+s + ".")
  
  var nun1 = Number(window.prompt("digite um número ")) //forma simplificada de somar no JavaScript
  var nun2 = Number(window.prompt("digite outro número para somar : ")) //forma mais simplificada de somar no JavaScript
  var s1 = nun1 + nun2
  window.alert("os números digitados foi "+ s1 + " <<<===--===>>> ")
  
  /* ############################## convertendo para string */
  
  window.alert("a forma de converter para string >>  "+ String(s1) + " <<")
  //convertendo o número para string 
  
</script>
</body>

</html>



```
